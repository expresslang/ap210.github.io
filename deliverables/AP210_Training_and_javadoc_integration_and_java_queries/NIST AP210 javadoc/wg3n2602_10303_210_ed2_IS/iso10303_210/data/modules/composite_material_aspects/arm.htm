 USE FROM Part_and_zone_laminate_tables_arm;    -- ISO/TS 10303-1770


TYPE compma_material_item_select = EXTENSIBLE GENERIC_ENTITY SELECT BASED_ON material_item_select WITH
   (Anisotropic_material,

    Woven_assembly);

END_ENTITY;
